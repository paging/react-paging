<div id="demo"></div>
var Paging = require('react-paging')
var React = require('react')
var App = React.createClass({
    getInitialState: function () {
        return {
            page: 10
        }
    },
    render: function () {
        var self = this
        return (
            <div>
                <Paging {...{
                    page: self.state.page,
                    pageCount: 20,
                    onChange: function (page) {
                        self.setState({
                            page: page
                        })
                    }
                }} />
            </div>
        )
    }
})

ReactDOM.render(<App />, document.getElementById('demo'))
